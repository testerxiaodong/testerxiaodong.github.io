class LRUCache:
    def __init__(self, capacity: int):
        self.cache = OrderedDict()
        self.capacity = capacity

    def get(self, key):
        if key not in self.cache:
            return -1
        self.cache.move_to_end(key)  # ÊúÄËøëËÆøÈóÆÁöÑÁßªÂä®Âà∞Êú´Â∞æ
        return self.cache[key]

    def put(self, key, value):
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.capacity:
            self.cache.popitem(last=False)  # ÁßªÈô§ÊúÄÊó©ÊèíÂÖ•ÁöÑÂÖÉÁ¥†ÔºàFIFOÔºâ

# Á§∫‰æã

cache.put(1, "A")
cache.put(2, "B")
print(cache.cache)  # OrderedDict([(1, 'A'), (2, 'B')])

cache.get(1)  # ËÆøÈóÆ 1Ôºå‰ΩøÂÖ∂ÂèòÊàêÊúÄËøë‰ΩøÁî®
cache.put(3, "C")  # Ê∑òÊ±∞ÊúÄ‰πÖÊú™‰ΩøÁî®ÁöÑ 2
